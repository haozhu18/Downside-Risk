df = pd.read_parquet(f'{_DATA_DIR}/ETFdata.parquet')
df = df[df['ticker'].isin(['FXY', 'FXE', 'FXF', 'FXB', 'FXA'])]
df_monthly = df.groupby(['ticker', 'ym'], as_index=False)['retM'].first()
df_monthly.rename(columns={'ym': 'ldate', 'retM': 'daret'}, inplace=True)
df_monthly.sort_values(by=['ldate', 'ticker'], inplace=True)
